<?php
session_start();

$db = new PDO('sqlite:data.sqlite');
$db->setAttribute(PDO::ATTR_ERRMODE, PDO::ERRMODE_EXCEPTION);

// Crear tabla usuarios si no existe
$db->exec("CREATE TABLE IF NOT EXISTS users (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    username TEXT UNIQUE,
    password TEXT,
    credits INTEGER DEFAULT 1000
)");

// Registro
if (isset($_POST['register'])) {
    $username = trim($_POST['username']);
    $password = password_hash($_POST['password'], PASSWORD_DEFAULT);

    $stmt = $db->prepare("INSERT INTO users (username, password) VALUES (?, ?)");
    try {
        $stmt->execute([$username, $password]);
        $_SESSION['user'] = $username;
        header("Location: index.php");
        exit();
    } catch (PDOException $e) {
        $error = "Usuario ya existe.";
    }
}

// Login
if (isset($_POST['login'])) {
    $username = trim($_POST['username']);
    $password = $_POST['password'];

    $stmt = $db->prepare("SELECT * FROM users WHERE username = ?");
    $stmt->execute([$username]);
    $user = $stmt->fetch(PDO::FETCH_ASSOC);

    if ($user && password_verify($password, $user['password'])) {
        $_SESSION['user'] = $username;
        header("Location: index.php");
        exit();
    } else {
        $error = "Usuario o contraseña incorrectos.";
    }
}

// Logout
if (isset($_GET['logout'])) {
    session_destroy();
    header("Location: index.php");
    exit();
}

// Usuario actual
$user = null;
if (isset($_SESSION['user'])) {
    $stmt = $db->prepare("SELECT * FROM users WHERE username = ?");
    $stmt->execute([$_SESSION['user']]);
    $user = $stmt->fetch(PDO::FETCH_ASSOC);
}

?>

<!DOCTYPE html>
<html lang="es">
<head>
<meta charset="UTF-8" />
<title>Panel Falix - Web simple</title>
<style>
body { font-family: Arial, sans-serif; background:#111; color:#eee; }
.container { max-width: 400px; margin: 40px auto; background:#222; padding:20px; border-radius:8px; }
input, button { width: 100%; margin: 8px 0; padding: 10px; border-radius:5px; border:none; }
button { background:#4CAF50; color:#fff; cursor:pointer; }
.error { color: #f44; }
.header { text-align:center; margin-bottom:20px; }
button.create-server { background:#2196F3; }
</style>
</head>
<body>
<div class="container">
    <div class="header"><h2>Panel Falix Simple</h2></div>

    <?php if ($error ?? false): ?>
        <p class="error"><?=htmlspecialchars($error)?></p>
    <?php endif; ?>

    <?php if (!$user): ?>
        <form method="POST">
            <input type="text" name="username" placeholder="Usuario" required />
            <input type="password" name="password" placeholder="Contraseña" required />
            <button type="submit" name="login">Iniciar Sesión</button>
            <button type="submit" name="register">Registrarse</button>
        </form>
    <?php else: ?>
        <p>Bienvenido, <strong><?=htmlspecialchars($user['username'])?></strong></p>
        <p>Créditos: <strong><?=htmlspecialchars($user['credits'])?></strong></p>
        <form method="POST" action="create_server.php">
            <button class="create-server" type="submit" name="create_server">Crear servidor</button>
        </form>
        <p><a href="?logout=1">Cerrar sesión</a></p>
    <?php endif; ?>
</div>
</body>
</html>
